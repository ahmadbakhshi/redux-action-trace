## Install

```bash
npm install --save redux-action-trace
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'redux-action-trace'
import 'redux-action-trace/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
